const CPF = {
    limpar(cpf) {
        return cpf.replace(/\D/g, "");
    },
    validar(cpf) {
        cpf = this.limpar(cpf);
        if (cpf.length !== 11) return false;
        if (/^(\d)\1+$/.test(cpf)) return false;

        let soma = 0;
        for (let i = 0; i < 9; i++) {
            soma += cpf[i] * (10 - i);
        }

        let dig1 = (soma * 10) % 11;
        if (dig1 === 10) dig1 = 0;

        soma = 0;
        for (let i = 0; i < 10; i++) {
            soma += cpf[i] * (11 - i);
        }

        let dig2 = (soma * 10) % 11;
        if (dig2 === 10) dig2 = 0;

        return dig1 == cpf[9] && dig2 == cpf[10];
    }
};
