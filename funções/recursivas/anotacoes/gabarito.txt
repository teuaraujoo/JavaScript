function somarAninhado(arr) {
    let soma = 0;

    for (const item of arr) {
        if (Array.isArray(item)) {
            soma += somarAninhado(item);
        } else {
            soma += item;
        }
    }

    return soma;
}

function contarProps(obj) {
    let total = 0;

    for (const chave in obj) {
        total++;
        if (typeof obj[chave] === 'object' && obj[chave] !== null) {
            total += contarProps(obj[chave]);
        }
    }

    return total;
}

const dados = {
    nome: "Ana",
    endereco: {
        rua: "A",
        cidade: "B"
    }
};

console.log(contarProps(dados));
