function criarCarrinho() {
    return {
        itens: [],
        adicionarItem(nome, preco) {
            this.itens.push({ nome, preco });
        },
        total() {
            let soma = 0;
            for (const item of this.itens) {
                soma += item.preco;
            }
            return soma;
        }
    };
}

function criarSistemaLogin(usuario, senha) {
    return {
        usuario,
        senha,
        logado: false,
        login(user, pass) {
            if (user === this.usuario && pass === this.senha) {
                this.logado = true;
                return true;

            return false;

        logout() {
            this.logado = false;
        }
    };
}

function criarProdutoValido(nome, validade) {
    return {
        nome,
        validade,
        estaValido() {
            const hoje = new Date();
            const [mes, ano] = this.validade.split('/');
            const dataValidade = new Date(`20${ano}`, mes, 0);
            return dataValidade >= hoje;
        }
    };
}

const produto = criarProdutoValido("Leite", "08/27");
console.log(produto.estaValido());

function criarContaBancaria(titular) {
    return {
        titular,

        }
    };
